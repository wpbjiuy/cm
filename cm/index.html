目录树

使用方法：var dirObj = cm.fnDirList(obj, callback).click(function(selectedData){})

	obj => {
		get:{
			url:'请求路径',
			setting:'请求数据',
			dataPath:'数据Key'
		},
		ttl:'头标',							String
		cls:'控制器',							Araay{css:'class',txt:'name',fn:function(){}}
		addDir:'添加目录',						Funtion Or Objecte => {url:'新增接口地址',dataType:'数据类型'}
		data:'数据',  							Array{}
		nextData:'下一级数据Key',  					Objecte => {dataPath:'数据Key',dirName:'显示名称'}
		id:'数据库Id',							String
		dirName:'目录名称Key',						String
		addDom:'添加列表层', 						Element
		css:'主体样式',							String
		href:'目录链接',						function(item){ return value; }
		on:0								Number Or Array([0,0])
	}

	dirObj => {
		updateAll:function(data){...},		//传入数据更新整个数据
		update:function(type,thsLi,name){...}, //更新单条数据 type = String('更新 = update'||'删除 = delete')， thsLi = Element(当前元素节点)，name = String(更新的内容)
		deleteList:function(thsLi){...},	//删除节点及数据  thsLi = Element(当前元素节点)
		updateList:function(thsLi,name){...},	//更新节点内容
		getOnData:function(path){...},		//根据数据路径获取数据 path = Number Or String(0-0-0)
		listOn:function(_ths){...},		//设置当前节点， _ths = Element(a 节点)
		addList:function(data, addDom, dirName){...},  //添加数据 data = Object(添加的数据)，addDom = Element(添加层节点)，dirName = String(显示数据Key)
		getData:function(obj){...}	//从新获取数据  obj = obj.get
	}

	obj => {

			setting:'请求参数',

			dataPath:'内容数据参数',
			thsData:'页面操作数据'
		},
		th:{ 			//表格表头
			name:'数据名称',
			txt:'头标名称',
			style:'sytle',
			filter:'数据过滤 function(数据){ return 处理后的数据;}> '
		},
		dataId:'数据ID',
		csle:[	//控制器
			{
				css:'样式类',
				icon:'图标样式类',
				txt:'名称',
				fnCk:'点击事件触发函数 function'
			}
		],
		csleType:'控制器类型 btn Or menu',
		tbCss:'表格样式',
		addDom:'添加层',

		Noc:'序号系数',

	}

	tbObj => {
		setAg:function(obj){...},	//从新设计GET请求的setting
		fnGetData:function(getSetting,isNew),  //请求数据
		resetShow:function(th){...},	//从新传入th对象，来显示对应的信息
		search:function(txt){...},	//查询表格数据，根据数据来查询
		updateList:function(data, isExtend){...}, //更新当前数据, data = Object(数据), isExtend = Boolean(是否同步数据)
		deleteList:function(){...}, //删除当前数据, 点击操作按钮就会绑定当前数据
		addList:function(data){...},  //添加列，  data = Object(数据)
		success:function(fn){...},  //设置GET请求成功后进行的操作， fn = Function(操作的函数)
		over:function(fn){...}		//设置表格生成结束后的操作， fn = Function(操作的函数)
	}

	arObj => [{节点一},{节点二}] -> Array[0] -> {

		getData:{url:'',dKey:'可选，数据存放的key',setting:{传递后台的参数}},    //后台请求数据 与 data 二选一

消息框

使用方法：提示框：cm.fnAlert(txt,hint,callback), 确认框：cm.fnConfirm(txt,callback), 右下角提示框：cm.fnFixedHint(txt) 

	txt => 消息内容,
	hint => 消息类型，'ok' Or 'err'
	callback => 回调函数
